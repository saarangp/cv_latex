\section*{Selected Research Projects}

\begin{projectlong}{Laya: EEG Foundation Model}{2024--2026}
    \item Developed Laya, the first JEPA-based EEG foundation model, showing that latent prediction yields more transferable, noise-robust representations than reconstruction-based pretraining.
    \item Pretrained on 30K hours ($>$ 1B tokens) across 20K+ subjects and 17 montages; evaluated on 14 BCI and clinical tasks via frozen linear probing.
    \item Controlled ablations isolate the pretraining objective as the primary driver of gains over reconstruction baselines.
\end{projectlong}

\begin{projectlong}{Thalamocortical Seizure Dynamics}{2023--2026}
    \item Characterized frequency-specific, directed thalamocortical connectivity across the full seizure trajectory in 19 pediatric epilepsy patients using SEEG.
    \item Showed that slow-frequency thalamic outflow to cortex exceeds cortico-cortical outflow during seizures, with nucleus-specific recruitment matching known anatomy (AN for limbic, CM for frontal).
    \item Thalamic spectral and connectivity features alone classify seizure state with AUROC 0.83--0.84.
\end{projectlong}

\begin{projectlong}{Seizure World Model}{2025--Present}
    \item Building a world model for seizure dynamics that encodes intracranial EEG into a learned latent space and predicts brain-state trajectories conditioned on stimulation parameters.
    \item Counterfactual rollouts forecast stimulation outcomes for closed-loop neurostimulation planning.
\end{projectlong}
